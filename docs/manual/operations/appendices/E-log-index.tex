\chapter{日志消息分类索引（运维视角）}

本附录从运维角度组织 \openbench{} 关键日志消息：哪些需要告警、哪些可抑制、哪些转发到聚合系统。

\noteinline{用户卷附录 E 也有错误消息列表，但视角不同：那里面向"我看到这条错怎么修"；本附录面向"我作为管理员要建监控告警，怎么设阈值"。}

\section{告警分级建议}

\subsection{紧急告警（P1）}

\textbf{触发}：服务/评估\emph{完全停止}。立即处理。

\begin{itemize}
  \item 任何 \texttt{CRITICAL} 级消息
  \item \texttt{[ERROR] OOMKilled}（如能捕获）
  \item \texttt{[ERROR] No space left on device}
  \item \texttt{[ERROR] Permission denied} 在 output\_dir
  \item Process exit code 非 0 且非 1（异常）
\end{itemize}

\subsection{严重告警（P2）}

\textbf{触发}：评估部分失败但程序继续。当天处理。

\begin{itemize}
  \item \texttt{[ERROR] task failed}（按 task 计数 > 5\%）
  \item \texttt{[ERROR] Reference resolution failed}
  \item \texttt{[ERROR] HDF5 lock} 复发（已修但偶现）
  \item \texttt{Evaluation completed with errors}
\end{itemize}

\subsection{告警（P3）}

\textbf{触发}：可疑但不阻塞。当周处理。

\begin{itemize}
  \item \texttt{[WARN] Deprecated} 频繁出现
  \item \texttt{[WARN] Provenance LOW} 大量出现（数据集元信息可疑）
  \item \texttt{[WARN] Cache invalidated unexpectedly}
\end{itemize}

\subsection{抑制（不告警）}

\textbf{常见但无害}：

\begin{itemize}
  \item \texttt{[INFO] Cache hit}
  \item \texttt{[INFO] Phase: ...}
  \item \texttt{[INFO] Output written}
  \item \texttt{[WARN] PDF generation skipped (xhtml2pdf not installed)}
  \item \texttt{[INFO] HDF5\_USE\_FILE\_LOCKING set to FALSE}
\end{itemize}

\section{按 logger 分类}

\subsection{openbench.cli.*}

CLI 入口；少量 INFO/ERROR：

\begin{longtable}{p{6cm} p{2cm} p{4cm}}
  \toprule
  消息 & 级别 & 处置 \\
  \midrule
  \endhead
  \texttt{Config error: ...} & ERROR & P3 用户问题 \\
  \texttt{Migration failed: ...} & ERROR & P3 \\
  \texttt{Remote execution not yet implemented} & WARN & 抑制 \\
  \bottomrule
\end{longtable}

\subsection{openbench.config.*}

Config 加载与翻译：

\begin{longtable}{p{6cm} p{2cm} p{4cm}}
  \toprule
  消息 & 级别 & 处置 \\
  \midrule
  \endhead
  \texttt{Missing required ...} & ERROR & P3 \\
  \texttt{Deprecated 'options' section} & WARN & 抑制（用户卷文档建议迁移） \\
  \texttt{!include file not found} & ERROR & P2 \\
  \texttt{tim\_res default — not confirmed} & WARN & 抑制（用户决定） \\
  \bottomrule
\end{longtable}

\subsection{openbench.runner.*}

主流程：

\begin{longtable}{p{6cm} p{2cm} p{4cm}}
  \toprule
  消息 & 级别 & 处置 \\
  \midrule
  \endhead
  \texttt{Starting evaluation: ...} & INFO & 抑制 \\
  \texttt{Cache hit: skipping ...} & INFO & 抑制 \\
  \texttt{Task failed: ...} & ERROR & P2 \\
  \texttt{Resource temporarily unavailable} & ERROR & P2（HDF5 lock） \\
  \texttt{Evaluation complete} & INFO & 抑制 \\
  \texttt{Evaluation failed} & ERROR & P1 \\
  \bottomrule
\end{longtable}

\subsection{openbench.data.*}

数据加载与处理：

\begin{longtable}{p{6cm} p{2cm} p{4cm}}
  \toprule
  消息 & 级别 & 处置 \\
  \midrule
  \endhead
  \texttt{Loading reference: ...} & INFO & 抑制 \\
  \texttt{No NC files matching pattern} & ERROR & P2（数据缺失） \\
  \texttt{Variable X not found in NC file} & ERROR & P3 \\
  \texttt{TimeAlignmentError} & ERROR & P3 \\
  \texttt{Skipping station: ...} & WARN & 抑制（按比例统计） \\
  \texttt{Regridding failed} & ERROR & P2 \\
  \bottomrule
\end{longtable}

\subsection{openbench.core.*}

评估引擎：

\begin{longtable}{p{6cm} p{2cm} p{4cm}}
  \toprule
  消息 & 级别 & 处置 \\
  \midrule
  \endhead
  \texttt{Computing metrics for ...} & INFO & 抑制 \\
  \texttt{Insufficient overlap years} & WARN & P3（按比例） \\
  \texttt{Empty dataset} & WARN & P3 \\
  \bottomrule
\end{longtable}

\subsection{openbench.gui.*}

GUI（仅 GUI 用户相关）：

\begin{longtable}{p{6cm} p{2cm} p{4cm}}

\end{longtable}

\section{建议的 Loki / ELK 查询}

\subsection{当前 run 错误数}

\begin{minted}{text}
{job="openbench"} |~ "\\[ERROR\\]" | json
\end{minted}

\subsection{HDF5 lock 复发监控}

\begin{minted}{text}
sum(count_over_time({job="openbench"} |~ "HDF5.*lock" [1h]))
\end{minted}

\subsection{Provenance LOW 比例}

\begin{minted}{text}
sum(count_over_time({job="openbench"} |~ "Provenance LOW" [1h]))
/
sum(count_over_time({job="openbench"} |~ "\\[INFO\\] Loading reference" [1h]))
\end{minted}

\section{Slack / Email 模板}

\subsection{P1 紧急}

\begin{minted}{text}
[OPENBENCH P1] Evaluation failed at <time>
- Run: <output_dir>
- Last error: <error message>
- Action: SSH to <host>, check df -h, free, dmesg
\end{minted}

\subsection{P2 严重}

\begin{minted}{text}
[OPENBENCH P2] Partial failure (<n> tasks failed) at <time>
- Run: <output_dir>
- Failed tasks:
  - <var> sim=<X> ref=<Y>
  - ...
\end{minted}
