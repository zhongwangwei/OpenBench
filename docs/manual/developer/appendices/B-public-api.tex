\chapter{Public Python API}

本附录列出可从外部脚本/Notebook 安全 import 的 \openbench{} 公开 API。破坏向后兼容会算 breaking change。

\section{顶层}

\begin{minted}{python}
import openbench
openbench.__version__   # str, e.g., '3.0.0a1'
\end{minted}

\section{config 子包}

\begin{minted}{python}
from openbench.config import (
    load_config,           # (path: str | Path) -> OpenBenchConfig
    ConfigError,           # Exception
    OpenBenchConfig,       # 顶层 schema dataclass
    ProjectConfig,
    EvaluationConfig,
    ReferenceConfig,
    SimulationEntry,
    ComparisonConfig,
    StatisticsConfig,
)
\end{minted}

\subsection{load\_config}

\begin{itemize}
  \item 签名：\texttt{load\_config(path: str | Path) -> OpenBenchConfig}
  \item 功能：读 YAML（支持 \texttt{!include}）、校验 schema、应用 backward compat
  \item 异常：\texttt{ConfigError}（路径不存在、YAML 解析失败、schema 不合法）
\end{itemize}

\subsection{OpenBenchConfig}

顶层容器；详细字段见附录 C 与 \file{config/schema.py}。

\section{runner}

\begin{minted}{python}
from openbench.runner.local import run_evaluation

results = run_evaluation(
    cfg,                # OpenBenchConfig
    force=False,        # bool, bypass incremental cache
    comparison_only=False,  # bool, skip evaluation phase
)
# results: dict with keys "status", "output_dir", "variables",
# "simulations", "errors"
\end{minted}

\section{registry}

\begin{minted}{python}
from openbench.data.registry import RegistryManager
from openbench.data.registry.manager import get_registry, get_writable_reference_catalog_path, get_writable_model_catalog_path

# 单例（推荐）
mgr = get_registry()

# 列出
mgr.list_references()  # list[ReferenceDataset]
mgr.list_models()       # list[ModelProfile]

# 按变量过滤
mgr.references_for_variable("Evapotranspiration")

# 按名查
mgr.get_reference("GLEAM_v4.2a_LowRes")  # ReferenceDataset | None
mgr.get_model("CoLM2024")                  # ModelProfile | None

# 分辨率变体
mgr.get_resolution_variants("GLEAM_v4.2a")
# {'LowRes': ReferenceDataset(...), 'MidRes': ReferenceDataset(...)}
\end{minted}

\subsection{Evaluation\_grid / Evaluation\_stn}

直接构造（不通过 runner）评估：

\begin{minted}{python}
from openbench.core import Evaluation_grid

evaluator = Evaluation_grid(info_dict)
result = evaluator.run()
\end{minted}

\texttt{info\_dict} 是 adapter 输出的 legacy 字典；详见 \file{config/adapter.py}。

\subsection{create\_evaluation\_engine（工厂）}

\begin{minted}{python}
from openbench.core import create_evaluation_engine

engine = create_evaluation_engine(data_type="grid", info=info_dict)
result = engine.evaluate()
\end{minted}

\section{使用示例}

\subsection{程序化跑评估}

\begin{minted}{python}
import os
os.environ["HDF5_USE_FILE_LOCKING"] = "FALSE"

from openbench.config import load_config
from openbench.runner.local import run_evaluation

cfg = load_config("openbench.yaml")
cfg.project.num_cores = 8

results = run_evaluation(cfg)

if results["status"] == "success":
    print(f"Done! Output: {results['output_dir']}")
else:
    for err in results["errors"]:
        print(f"  - [{err['phase']}] {err['message']}")
\end{minted}

\subsection{直接调 metric}

\begin{minted}{python}
import numpy as np
from openbench.core import metrics

m = metrics()
sim = np.array([1.0, 2.1, 2.9, 4.0])
obs = np.array([1.0, 2.0, 3.0, 4.0])

print(f"bias: {m.bias(sim, obs):.3f}")
print(f"KGE:  {m.KGE(sim, obs):.3f}")
\end{minted}

\subsection{遍历 registry}

\begin{minted}{python}
from openbench.data.registry.manager import get_registry

mgr = get_registry()
for ref in mgr.list_references():
    if ref.category == "Water":
        print(f"{ref.name}: {ref.tim_res} {ref.grid_res}")
\end{minted}

\section{API 稳定性承诺}

\begin{itemize}
  \item \texttt{load\_config} 签名稳定；返回的 dataclass 字段可能新增（不删）
  \item \texttt{run\_evaluation} 签名稳定；返回 dict 的 key 可能新增（不删）
  \item \texttt{metrics} / \texttt{scores} 已有方法签名稳定；可能新增方法
  \item \texttt{RegistryManager} 公开方法稳定
  \item visualization 函数签名稳定（已暴露在 \texttt{\_\_all\_\_}）
\end{itemize}

\warninline{私有成员（带下划线）、\modname{config.adapter}、\modname{config.resolver}、\modname{runner.cache} 等内部接口随时变；如要 import 请提 issue 申请提升为公开 API。}
